📚 고급 개념

Guard Clause Pattern

function process(data) {
    if (!data) return null;
    if (!data.isValid) return false;
    if (data.type !== 'expected') return;
    
    // 핵심 로직
    return processData(data);
}

Short Circuit Evaluation

// AND (&&) - 첫 번째 falsy 값 반환
const result = value && value.property;

// OR (||) - 첫 번째 truthy 값 반환
const fallback = userInput || defaultValue;

// Nullish Coalescing (??)
const safe = value ?? fallbackValue;

Object Lookup Pattern

const handlers = {
    'type1': () => handleType1(),
    'type2': () => handleType2(),
    'default': () => handleDefault()
};

const handler = handlers[type] || handlers.default;
handler();

Strategy Pattern

const strategies = {
    premium: (price) => price * 0.8,
    regular: (price) => price * 0.95,
    new: (price) => price * 0.9
};

const discount = strategies[userType](price);

Complex Nested Conditions

if (user.isActive) {
    if (user.hasPermission) {
        if (resource.isAvailable) {
            // 모든 조건 충족
        }
    }
}

// Better approach:
const canAccess = user.isActive 
    && user.hasPermission 
    && resource.isAvailable;

⚡ 성능 최적화 팁

  • 조건 순서 최적화: 가장 빨리 실패할 조건을 먼저 체크
  • Early Return: 불필요한 중첩을 피하고 조기 반환 활용
  • 캐싱: 반복적인 계산 결과는 변수에 저장
  • Object Lookup: 많은 if-else 대신 객체 매핑 사용
  • 단축 평가: && 와 || 연산자의 특성 활용
  • Optional Chaining: ?. 연산자로 안전한 프로퍼티 접근
