📖 조건문 개념

if문

if (조건) {
    // 조건이 true일 때 실행
}

if-else문

if (조건) {
    // 조건이 true일 때
} else {
    // 조건이 false일 때
}

else if문

if (조건1) {
    // 조건1이 true일 때
} else if (조건2) {
    // 조건2가 true일 때
} else {
    // 모든 조건이 false일 때
}

복합 조건

// AND 연산자 (&&)
if (조건1 && 조건2) {
    // 두 조건 모두 true일 때
}

// OR 연산자 (||)
if (조건1 || 조건2) {
    // 하나라도 true일 때
}

switch문

switch(변수) {
    case 값1:
        // 처리
        break;
    case 값2:
        // 처리
        break;
    default:
        // 기본 처리
}

삼항 연산자

const 결과 = 조건 ? 참일때값 : 거짓일때값;
